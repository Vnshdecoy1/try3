```typescript
  {tool.pricing.map(tier => (
    <div onClick={() => setSelectedPricingTier(tier)}>

    </div>
  ))}

```

### Custom Input Tools  
```typescript
// Real-time calculation
const price = inputValue * pricePerUnit;
setSelectedPackage({
  tier: `${inputValue} ${unitName}`,

  amount: inputValue
});
```

```typescript

setSelectedBumpSpeed(speedTier);
```
